  const videoBtns = document.querySelectorAll(".video__button");
  const videosPanels = document.querySelectorAll(".video__control"); 
  const videos = document.querySelectorAll("video");

  videoBtns.forEach((videoBtn, index) => {
    videoBtn.addEventListener("click", function (evt) {
      evt.preventDefault();
      videos[index].play();
      videos[index].setAttribute("controls", "");
      videosPanels[index].classList.add("video__control--active");
    });
  });

  videos.forEach((video, index) => {
    video.addEventListener("pause", function (evt) {
      this.removeAttribute("controls");
      videosPanels[index].classList.remove("video__control--active");
    });
  });
