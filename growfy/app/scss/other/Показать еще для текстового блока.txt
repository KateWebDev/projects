JS

 const postItems = document.querySelector(".post__items");
  const postBtn = document.querySelector(".post__btn");

  const textBtn = postBtn.textContent;
  const heightItems = postItems.offsetHeight;

  postBtn.addEventListener("click", function (evt) {
    evt.preventDefault();
    if (postItems.offsetHeight < postItems.scrollHeight) {
      postItems.style.maxHeight = `${postItems.scrollHeight / 16}rem`;
      this.textContent = "Hide text";
      postItems.classList.add("post__items--active");
      this.classList.add("post__btn--active");
    } else {
      postItems.style.maxHeight = `${heightItems / 16}rem`;
      postItems.classList.remove("post__items--active");
      this.textContent = textBtn;
      this.classList.remove("post__btn--active");
    }
  });
